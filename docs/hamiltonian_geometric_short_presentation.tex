% ---------- 2. Core Question ----------
\begin{frame}{Core Question}
  \begin{block}{Goal}
    Can a single \textbf{Hamiltonian-geometric} update rule act as a useful optimizer family
    across classical ML, LLM-style training, PINNs, and quantum-control objectives?
  \end{block}
  \begin{itemize}
    \item Treat parameters $\theta$ as generalized coordinates, the loss $L(\theta)$ as potential
      energy, and optimizer state as momentum $p$ evolving on a \textbf{metric manifold} $g(\theta)$.
    \item One family recovers SGD, momentum, Adam, and natural gradient as special cases.
    \item The claim is deliberately \emph{bounded}: not uniformly superior, strongest when
      full-metric curvature is exploitable --- and we test that boundary directly, not just assert it.
  \end{itemize}
\end{frame}

% ---------- 4. Mathematical framework: equations ----------
\begin{frame}{The Update Rule --- and Why Adam and SGD Are Special Cases of It}
  \small
  \begin{align*}
    H(\theta,p) &= \tfrac{1}{2}p^{\top}g^{-1}(\theta)p + L(\theta),
    \qquad \dot\theta = \partial H/\partial p, \quad \dot p = -\partial H/\partial \theta \\
    p_{t+1} &= \beta p_t - \eta\big[\nabla_\theta L + F_{\mathrm{geo}} + F_{\mathrm{mem}} + \alpha\nabla_\theta S(g)\big],
    \qquad \theta_{t+1} = \theta_t + \eta\, g^{-1} p_{t+1}
  \end{align*}
  \vspace{-0.3em}
  \scriptsize
  \begin{tabular}{@{}p{0.22\linewidth}p{0.7\linewidth}@{}}
    \toprule
    Optimizer & Hamiltonian-geometric specialization \\
    \midrule
    SGD / heavy-ball & Flat metric, $F_{\mathrm{geo}}=0$ \\
    Adam / RMSProp & Diagonal adaptive metric; ignores metric-variation force \\
    Natural gradient & Full information metric, no momentum/memory force \\
    \textbf{Hamiltonian-geometric} & Full metric \emph{with} momentum, memory, spectral entropy, curvature \\
    \bottomrule
  \end{tabular}
\end{frame}
